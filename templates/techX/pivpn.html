Noip installation


cd /usr/local/src/
wget http://www.noip.com/client/linux/noip-duc-linux.tar.gz
tar xf noip-duc-linux.tar.gz
cd noip-2.1.9-1/
sudo make install
###
- etho0
- email: noip email
- password: noip pwd
- update interval: 60 (mins)
- run sth at successful update: n

Run noip and enable


ps ax | grep noip2 # noip2 not running
sudo noip2 # running but not enable
sudo vim /etc/systemd/system/noip2.service # edit systemd file
###
[Unit]
Description=No-ip.com dynamic IP address updater
After=network.target
After=syslog.target

[Install]
WantedBy=multi-user.target
Alias=noip.service

[Service]
ExecStart=/usr/local/bin/noip2
Restart=always
Type=forking
###
sudo systemctl enable noip2
sudo systemctl start noip2
sudo ufw allow 1194
# Then port forwarding and firewall 1194 on router #############
sudo reboot
